# ビュー構成設計書

> **最終更新**: 2025年12月17日
>
> 本ドキュメントは、スタッフ・家族間でデータが交差するビュー構成を定義します。
> 両者が同じデータを異なる視点で閲覧・操作できる整合性のある設計を目指します。

---

## 1. 概要

### 1.1 設計原則

1. **データ交差**: 家族が登録した品物をスタッフが閲覧・操作し、その結果を家族が確認
2. **共有ビュー**: 記録閲覧・統計など、同じデータを両者が見れるビュー
3. **専用ビュー**: 入力・操作系はロール専用（誤操作防止）
4. **リアルタイム同期**: 更新があれば即座に反映

### 2.2 App.tsx ルーティング実装

```tsx
// frontend/src/App.tsx
import { Routes, Route, Navigate } from 'react-router-dom';

// 共有ビュー
import { ViewPage } from './pages/ViewPage';
import { StatsDashboard } from './pages/shared/StatsDashboard';
import { ItemTimeline } from './pages/shared/ItemTimeline';

// スタッフ専用
import { StaffHome } from './pages/staff/StaffHome';
import { MealInputPage } from './pages/MealInputPage';
import { FamilyMessages } from './pages/staff/FamilyMessages';
import { FamilyMessageDetail } from './pages/staff/FamilyMessageDetail';

// 家族専用
import { FamilyDashboard } from './pages/family/FamilyDashboard';
import { ItemManagement } from './pages/family/ItemManagement';
import { ItemForm } from './pages/family/ItemForm';
import { ItemDetail } from './pages/family/ItemDetail';
import { RequestBuilder } from './pages/family/RequestBuilder';
import { PresetManagement } from './pages/family/PresetManagement';
import { TaskList } from './pages/family/TaskList';
import { EvidenceMonitor } from './pages/family/EvidenceMonitor';

function App() {
  return (
    <Routes>
      {/* デフォルト: /view へリダイレクト */}
      <Route path="/" element={<Navigate to="/view" replace />} />

      {/* ========== 共有ビュー ========== */}
      {/* 記録閲覧（スタッフ・家族共通） */}
      <Route path="/view" element={<ViewPage />} />

      {/* 統計ダッシュボード */}
      <Route path="/stats" element={<StatsDashboard />} />

      {/* 品物タイムライン（共有） */}
      <Route path="/items/:id/timeline" element={<ItemTimeline />} />

      {/* ========== スタッフ専用 ========== */}
      {/* スタッフホーム */}
      <Route path="/staff" element={<StaffHome />} />

      {/* 食事記録入力 */}
      <Route path="/staff/input/meal" element={<MealInputPage />} />

      {/* 家族連絡一覧 */}
      <Route path="/staff/family-messages" element={<FamilyMessages />} />

      {/* 家族連絡詳細（消費記録入力） */}
      <Route path="/staff/family-messages/:id" element={<FamilyMessageDetail />} />

      {/* スタッフ用統計（共有ページへリダイレクト） */}
      <Route path="/staff/stats" element={<Navigate to="/stats" replace />} />

      {/* ========== 家族専用 ========== */}
      {/* 家族ホーム */}
      <Route path="/family" element={<FamilyDashboard />} />

      {/* 品物管理 */}
      <Route path="/family/items" element={<ItemManagement />} />
      <Route path="/family/items/new" element={<ItemForm />} />
      <Route path="/family/items/:id" element={<ItemDetail />} />

      {/* ケア指示 */}
      <Route path="/family/request" element={<RequestBuilder />} />
      <Route path="/family/request/:id" element={<RequestBuilder />} />

      {/* プリセット管理 */}
      <Route path="/family/presets" element={<PresetManagement />} />

      {/* タスク一覧 */}
      <Route path="/family/tasks" element={<TaskList />} />

      {/* エビデンス確認 */}
      <Route path="/family/evidence/:date" element={<EvidenceMonitor />} />

      {/* 家族用統計（共有ページへリダイレクト） */}
      <Route path="/family/stats" element={<Navigate to="/stats" replace />} />

      {/* レガシーパス（後方互換） */}
      <Route path="/input/meal" element={<Navigate to="/staff/input/meal" replace />} />
      <Route path="/sheet/:sheetName" element={<ViewPage />} />
    </Routes>
  );
}

export default App;
```

### 3.3 フッター切り替えロジック

```typescript
// frontend/src/components/FooterNav.tsx

export function FooterNav() {
  const location = useLocation();

  // パスでロールを判定
  const isFamilyPath = location.pathname.startsWith('/family');
  const isStaffPath = location.pathname.startsWith('/staff');

  // 共有ビュー（/view, /stats, /items）はコンテキストを維持
  // 直前のパスを記憶してロールを判定
  const savedRole = localStorage.getItem('userRole') || 'staff';
  const isFamily = isFamilyPath || (!isStaffPath && savedRole === 'family');

  if (isFamily) {
    return <FamilyFooter />;
  }

  return <StaffFooter />;
}

function StaffFooter() {
  const location = useLocation();

  return (
    <nav className="footer-nav">
      <NavLink
        to="/view"
        className={({ isActive }) => isActive ? 'active' : ''}
      >
        <ClipboardIcon />
        <span>記録閲覧</span>
      </NavLink>

      <NavLink
        to="/staff/input/meal"
        className={({ isActive }) => isActive ? 'active' : ''}
      >
        <PencilIcon />
        <span>記録入力</span>
      </NavLink>

      <NavLink
        to="/staff/family-messages"
        className={({ isActive }) =>
          isActive || location.pathname.startsWith('/staff/family-messages')
            ? 'active'
            : ''
        }
      >
        <UsersIcon />
        <span>家族連絡</span>
      </NavLink>

      <NavLink
        to="/stats"
        className={({ isActive }) => isActive ? 'active' : ''}
      >
        <ChartBarIcon />
        <span>統計</span>
      </NavLink>
    </nav>
  );
}

function FamilyFooter() {
  const location = useLocation();

  return (
    <nav className="footer-nav">
      <NavLink
        to="/family"
        end
        className={({ isActive }) => isActive ? 'active' : ''}
      >
        <HomeIcon />
        <span>ホーム</span>
      </NavLink>

      <NavLink
        to="/family/items"
        className={({ isActive }) =>
          isActive || location.pathname.startsWith('/family/items')
            ? 'active'
            : ''
        }
      >
        <CubeIcon />
        <span>品物管理</span>
      </NavLink>

      <NavLink
        to="/view"
        className={({ isActive }) => isActive ? 'active' : ''}
      >
        <ClipboardIcon />
        <span>記録閲覧</span>
      </NavLink>
```
